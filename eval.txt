COMANDOS IMPORTANTES PARA DEFESA DO PROJETO INCEPTION

[GERAL / MAKEFILE]
make
make build
make up
make down
make clean
make fclean
make re

[DOCKER BÁSICO]
docker ps
docker ps -a
docker images
docker volume ls
docker network ls

[ACESSAR CONTAINERS]
docker exec -it nginx bash
docker exec -it wordpress bash
docker exec -it mariadb bash

[LOGS DOS SERVIÇOS]
docker logs nginx
docker logs wordpress
docker logs mariadb

[NGINX – ETAPA: CONFIGURAÇÃO E DEPURAÇÃO]
nginx -t
curl -k https://localhost
openssl x509 -in /etc/ssl/certs/certificate.crt -text

[WORDPRESS / PHP-FPM – ETAPA: TESTE DO FASTCGI]
php-fpm8.2 -t

[MARIADB – ETAPA: TESTE DO BANCO]
mysql -u "$MYSQL_USER" -p
mysqladmin ping -h localhost

[INSPEÇÃO E DEBUG – ETAPA FINAL]
docker inspect nginx
docker inspect wordpress
docker inspect mariadb
docker-compose config
docker-compose top
docker-compose ps

[PARAR E REMOVER – LIMPEZA DO PROJETO]
docker stop nginx wordpress mariadb
docker rm nginx wordpress mariadb

[VERIFICAR VOLUMES – ETAPA DE PERSISTÊNCIA]
docker volume inspect wp_data
docker volume inspect db_data

[TESTES DE REDE – ETAPA DA NETWORK]
hostname -I
ping mariadb
ping wordpress
